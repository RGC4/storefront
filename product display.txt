    // Decode HTML entities that WordPress/Blade may encode into data attributes
    // (e.g. & → &amp; or &#038; — both end up as &#038; after double-encoding)
    function decodeComboUrl(url) {
        return url.replace(/&#0*38;/g, '&').replace(/&amp;/g, '&');
    }

    // Apply a combo's filter URL to InstantSearch without navigation.
    // Calls window.dxpApplyShopFilters (exposed by main.js), which runs
    // restoreUrlToHelper + helper.search() without firing popstate to other listeners.
    function dxpApplyComboUrl(comboUrl, activePill) {
        comboUrl = decodeComboUrl(comboUrl);
        var qs = comboUrl.indexOf('?') !== -1 ? comboUrl.split('?')[1] : '';

        // Update active pill state
        document.querySelectorAll('.dxp-combo-pill').forEach(function (p) {
            p.setAttribute('aria-pressed', p === activePill ? 'true' : 'false');
        });

        if (typeof window.dxpApplyShopFilters === 'function') {
            window.dxpApplyShopFilters(qs);
        }

        // Scroll to the products grid
        var row = document.querySelector('.row');
        if (row) row.scrollIntoView({ behavior: 'smooth', block: 'start' });
    }

    // Normalize a query string to a sorted, canonical form for comparison
    function normalizeQs(qs) {
        if (!qs || qs === '?') return '';
        var raw = qs.charAt(0) === '?' ? qs.slice(1) : qs;
        return raw.split('&').map(function (p) { return decodeURIComponent(p); }).sort().join('&');
    }

    // Mark the pill whose combo URL matches the current page URL on load
    function syncActivePill() {
        var currentNorm = normalizeQs(window.location.search);
        document.querySelectorAll('.dxp-combo-pill[data-combo-url]').forEach(function (pill) {
            var pillUrl = decodeComboUrl(pill.dataset.comboUrl);
            var pillQs = pillUrl.indexOf('?') !== -1 ? pillUrl.split('?')[1] : '';
            var pillNorm = normalizeQs(pillQs);
            pill.setAttribute('aria-pressed', pillNorm && pillNorm === currentNorm ? 'true' : 'false');
        });
    }

    // Pill click handler
    document.addEventListener('click', function (e) {
        var pill = e.target.closest('.dxp-combo-pill');
        if (!pill) return;
        var comboUrl = pill.dataset.comboUrl;
        if (!comboUrl) return;
        dxpApplyComboUrl(comboUrl, pill);
    });

    // Sync active state on load and after browser navigation
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', syncActivePill);
    } else {

            var isActive = pillQs && normalizeQs(pillQs) === currentNorm ? 'true' : 'false';
